62   const Home = ({onLoggedOut, currentUsername}) => {
63     const [currentScreen, setScreen] = useState("menu");
64     const [score, setScore] = useState(50);
65     const [hasPlayed, setPlayed] = useState(false);
66   
67     const [firebaseData, setFirebaseData] = useState([]);
68     const [firebaseLoaded, setFirebaseLoaded] = useState(false);
69   
70     const word = "hello";
71     const letters = word.split('');
72   
73     const [rows, setRows] = useState(new Array(NUMBER_OF_TRIES).fill(new Array(letters.length).fill("")));
74   
75     const [curRow, setCurRow] = useState(0);
76     const [curCol, setCurCol] = useState(0);
77     const [gameState, setGameState] = useState('playing');
78     const [data, setData] = useState([]);
79   
80     const onPress = () => {
81       // alert(printUsers)
82       if(true) {
83         setScreen("game");
84       }
85       else {
86         Alert.alert("YOU CANT", "you already played the game today")
87       }
88     }
89   
90     const onPress2 = () => {
91       setScreen("menu");
92     }
93     const onPresslboard = () => {
94       setScreen("lboard");
95     }
96   
97     useEffect(async () => {
98       if (curRow > 0) {
99         checkGameState();
100      }
101  
102      //initially loading the page - we run this piece of code and setting the
103      // flag - already loaded in order to prevent infinite loop
104      if (!firebaseLoaded) {
105        setFirebaseLoaded(true);
106        setFirebaseData(await getFakeFirestoreData());
107        //setScreen("lboard")
108  
109        //trying to fetch data from firebase - not working because firestore()
110        // is not configured
111        // firestore()
112        //   .collection('Players')
113        //   //.doc(userId)
114        //   .onSnapshot(documentSnapshots=> {
115        //     console.log('data: ', documentSnapshots.data());
116        //   });
117      }
118    })
119  
120    const checkGameState = () => {
121      if (checkIfWon()) {
122        // setScore(32 + (curRow)*14))
123        Alert.alert("GAME OVER", "You win")
124        setGameState('won')
125        setTimeout(() => {
126          clearBoard();
127        }, 1000);
128        setScreen("result")
129        setPlayed(true)
130      } else if (checkIfLost()) {
131        setScore(0)
132        Alert.alert("GAME OVER", "You lose")
133        setGameState('lost')
134        setTimeout(() => {
135          clearBoard();
136        }, 1000);
137        setScreen("result")
138        setPlayed(true)
139      }
140    }
141  
142    const clearBoard = () => {
143      const updatedRows = copyArray(rows);
144      for (var i = 0; i < 6; i++) {
145        for (var j = 0; j < 5; j++) {
146          updatedRows[i][j] = "";
147        }
148      }
149      setRows(updatedRows);
150      setCurRow(0);
151      setCurCol(0);
152      setGameState('playing')
153    }
154  
155    const checkIfWon = () => {
156      const row = rows[curRow - 1];
157      setScore(30 + ((6-curRow)*14))
158      return row.every((letter, i) => letter === letters[i])
159    }
160  
161    const checkIfLost = () => {
162      return curRow === rows.length;
163    }
164  
165    const onKeyPressed = (key) => {
166      if (gameState !== "playing") {
167        return;
168      }
169  
170      const updatedRows = copyArray(rows);
171  
172      if (key === CLEAR) {
173        const prevCol = curCol - 1;
174        if (prevCol >= 0) {
175          updatedRows[curRow][prevCol] = "";
176          setRows(updatedRows);
177          setCurCol(prevCol);
178        }
179        return;
180      }
181  
182      if (key === ENTER) {
183        if (curCol === rows[0].length) {
184  
185          const guess = rows[curRow].join("");
186          console.log(guess);
187          const url = "https://api.dictionaryapi.dev/api/v2/entries/en/" + guess;
188  
189          fetch(url)
190            .then((response) => response.json())
191            .then((responseJson) => {
192              console.log(responseJson);
193              if (responseJson["title"] === "No Definitions Found") {
194                Alert.alert("SORRY", "This word does not exist")
195                updatedRows[curRow][curCol - 1] = "";
196                setRows(updatedRows);
197                setCurCol(curCol - 1);
198              } else {
199                setCurRow(curRow+1);
200                setCurCol(0);
201              }
202            })
203            .catch((error) => {
204              console.error(error);
205            });
206  
207        }
208        return;
209      }
210  
211      if(curCol < rows[0].length) {
212        updatedRows[curRow][curCol] = key;
213        setRows(updatedRows);
214        setCurCol(curCol+1);
215      }
216    }
217  
218    const isCellActive = (row, col) => {
219      return row === curRow && col === curCol;
220    }
221  
222    const getCellBGColor = ( row, col) => {
223      const letter = rows[row][col];
224      if (row >=  curRow) {
225        return colors.black;
226      }
227      if (letter === letters[col]) {
228        return colors.primary;
229      }
230      if (letters.includes(letter)) {
231        return colors.secondary;
232      }
233      return colors.darkgrey;
234    }
235  
236    const getAllLettersWithColor = (color) => {
237      return rows.flatMap((row, i) => row.filter((cell, j) => getCellBGColor(i, j) === color));
238    }
239  
240    const greenCaps = getAllLettersWithColor(colors.primary)

281        </View>
282      );
283    }
284    else if(currentScreen === "game"){
285      return (
286        <View style={styles.container}>
287          <StatusBar style="Light" />

289          <Text style = {styles.title}>PUZLER</Text>

291          <View style={styles.map}>

293            {rows.map((row, i) => (
294              <View  key = {`row-${i}`}  style={styles.row}>
295                {row.map((letter, j) => (
296                  <View
297                    key = {`cell-${i}-${j}`}
298                    style={[styles.cell,
299                      {
300                        borderColor: isCellActive(i, j) ? colors.lightgrey: colors.darkgrey,
301                        backgroundColor: getCellBGColor(i, j)
302  
303                      }]
304                    }>
305  
306                    <Text style = {styles.cellText}>{letter.toUpperCase()}</Text>
307  
308                  </View>
309                ))}
310  
311              </View>
312            ))}
313  
314          </View>

316          <Keyboard onKeyPressed={onKeyPressed} greenCaps={greenCaps} yellowCaps={yellowCaps} greyCaps={greyCaps}/>
317        </View>
318      );
319    }
320    else if(currentScreen === "result") {
321      return (
322        <View style={styles.menuContainer}>
323          <View>
324            <Text style={styles.menuTittleText}>GOOD JOB</Text>
325          </View>
326          <View>
327            <Text style={styles.subscript}>you got a score of score {score}</Text>
328          </View>
329          <TouchableOpacity
330            style={styles.menuButton}
331            onPress={onPress2}
332          >
333            <Text style={styles.menuButtonText}>BACK TO MENU</Text>
334          </TouchableOpacity>
335  
336        </View>
337      );
338    }
339    //start of leaderboard rendering
340    else if(currentScreen === "lboard") {
341      return (
342        <View style={styles.menuContainer}>
343          <div className="board">
344            <h1 className='leaderboard'>Leaderboard</h1>
345  
346            <Profiles Leaderboard={firebaseData} />
347  
348            <Button
349              onPress={onReloadPressed}
350              title="RELOAD"
351              color="#841584"
352            />
353          </div>
354        </View>
355      );

357    else {
358      return (<div>UNKNOWN SCREEN</div>);
359    }
